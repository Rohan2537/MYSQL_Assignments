12)	DISPLAY THE NAMES, DESIGNATION AND SALARIES OF ALL EMPLOYEES WHO HAVE MANAGER ALONG WITH MANAGER'S NAME, DESIGNATION AND MANAGER'S SALARY.
	(SELF-JOIN)

query:	SELECT e.employee_id,e.first_name,e.job_id,e.salary,f.first_name AS manger_name,f.job_id AS designation,f.salary AS manager_salary FROM employees e INNER JOIN employees f ON e.manager_id=f.employee_id;
